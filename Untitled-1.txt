INSERT
INTO MWP_THREADABLEEVENT
  (
    THREADID,
    EVENTCATEGORYID,
    EVENTSTARTTIME,
    EVENTENDTIME,
    EVENTURL,
    EVENTPOSTSTARTTIME,
    EVENTPOSTENDTIME,
    TEMP_ID
  )
  
  
  
SELECT 
  THREADID,
  THREADCATEGORYID,
  THREADTITLE,
  THREADTIMESTAMP,
  THREADABLEAUTHOR,
  THREADCONTENT,
  THREADISPUBLISHED,
  TEMP_ID
FROM MWP_THREADABLE
WHERE THREADCATEGORYID = 3;









SELECT 
  curr.THREADID as NEW_THREAD_ID,
  CAL_NUM,
  CAL_CALENDAR_ID,
  CAL_STARTDATE,
  CAL_ENDDATE,
  CAL_EVENT_TYPE,
  USER_NAME,
  CAL_DESCRIPTION,
  CAL_URL,
  CAL_CATEGORY_ID,
  THREAD_ID,
  CAL_SUBJECT,
  CAL_DESCRIPTION_CLOB,
  CAL_EVENTSTART,
  CAL_EVENTEND,
  CAL_EVENT_STATUS
FROM PORTAL.TBL_CALENDAR_EVENT prev
LEFT JOIN MWP.MWP_THREADABLE curr ON prev.THREAD_ID =  curr.TEMP_ID;
